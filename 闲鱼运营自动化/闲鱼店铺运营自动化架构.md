class CustomerServiceAgent:
    def on_new_message(self, msg, customer):
        # 1. 加载客户历史
        history = self.db.get_history(customer.id)
        
        # 2. 意向度评分 (0-100)
        intent_score = self.score_intent(msg, history)
        
        # 3. RAG 检索
        context = self.vector_db.search(msg, top_k=5)
        
        # 4. 决策树
        if intent_score > 80:
            # 高意向 → 引流
            reply = self.llm.invoke(f"""
            客户消息: {msg}
            客户历史: {history}
            知识库: {context}
            
            意向度高，请生成引流话术，逻辑:
            1. 先回答他的问题
            2. 暗示需求复杂需要详聊
            3. 引导加微信 (用暗语:VX/v/微信)
            """)
            self.send_reply(reply)
            self.trigger_followup(customer, delay_hours=24)
            
        elif intent_score < 30:
            # 低意向 → 标准应答
            reply = self.llm.invoke(f"""
            客户消息: {msg}
            知识库: {context}
            生成简短礼貌回复，不要主动推销。
            """)
            self.send_reply(reply)
            
        else:
            # 中意向 → AI 多轮对话
            reply = self.llm.invoke(SYSTEM_PROMPT + history + msg)
            self.send_reply(reply)
        
        # 5. 涉及报价 → 必须人工
        if any(kw in msg for kw in ["多少钱", "便宜", "砍价", "议价"]):
            self.notify_human(customer, msg, suggested_reply=reply)
            return  # 等你确认再发

**意向度评分模型**：

```python
HIGH_INTENT_SIGNALS = [
    "什么时候能做完", "源码给吗", "可以加微信吗",
    "我有 X 个网站", "急用", "明天要", "怎么付款"
]

LOW_INTENT_SIGNALS = [
    "随便问问", "看看", "考虑一下", "太贵了"
]

def score_intent(msg, history):
    base = 50
    for sig in HIGH_INTENT_SIGNALS:
        if sig in msg: base += 15
    for sig in LOW_INTENT_SIGNALS:
        if sig in msg: base -= 10
    if len(history) > 5: base += 20  # 多轮对话
    if "?" in msg: base += 5  # 主动提问
    return min(100, max(0, base))
```
